Introduction class (simpleGE.Scene):
	set background image
	create -
		title = multilabel w/ title text, size of (500,300)
		btnPlay = set text "Play", position to (100,800)
		btnQuit = set text "Quit", position to (100,400)

	sprites: {title, btnPlay, btnQuit,}

	process():
		if quit clicked: play 'click sound', quit
		if start clicked: play 'click sound', start	


Desk class (simpleGE.Scene):
	init (size = (1280,720), support = 0, awareness = 0, survivors = 0, casualties = 0, turns_left = 0)
		super __init__(size)
		set background image

		set label variables -
			self.support = support
			self.awareness = awareness
			self.survivors = survivors
			self.casualties = casualties
			self.turns_left = turns_left

		create -
			lblSupport = simpleGE.Label()
			self.~.text = f"Support: {self.support}%"
			self.~.center = (120,57.5)
			self.~.size = (200,75)

			lblAwareness = simpleGE.Label()
			self.~.text = f"Awareness: {self.awareness}%"
			self.~.center = (120,157.5)
			self.~.size = (200,75)

			lblSurvivors = simpleGE.Label()
			self.~.text = f"Survivors: {self.survivors}"
			self.~.center = (345,57.5)
			self.~.size = (200,75)

			lblCasualties = simpleGE.Label()
			self.~.text = f"Casualties: {self.casualties}"
			self.~.center = (345,157.5)
			self.~.size = (200,75)

			lblTurnsLeft = simpleGE.Label()
			self.~.text = f"Turns Left: {self.turns_left}"
			self.~.center = (1116,57.5)
			self.~.size = (200,75)



			btnApprove = simpleGE.Button()
			text = "APPROVE"
			size = (200,75)
			position = ()
			color = green

			btnDeny = simpleGE.Button()
			text = "DENY"
			size = (200,75)
			position = ()
			color = red


		self.response = "remain_desk"
		self.document_A = Document_A(self)
		self.document_B = Document_B(self)


		sprites: {lblSupport, lblAwareness, lblSurvivors, lblCasualties, [document_sprites]}

		process(self):
		---
# switch to the map every 2 document decisions
			phase = 0

			if phase = 0:
				if btnApprove clicked:
					self.document_A.visible = False
					self.document_A.update_values()
					phase += 1

				if btnDeny clicked:
					self.document_A.visible = False
					self.document_B.visible
					phase += 1

			if phase = 1:
				if btnApprove clicked:
					self.document_B.visible = False
					self.document_B.update_values()
					phase += 1

				if btnDeny clicked:
					self.document_B.visible = False
					phase += 1

			if phase = 2:
				response = "go_to_map"
				stop
			


		---
